\begin{exo}[Une généralisation du point fixe] 
Soit $(E,d)$ un espace métrique complet non vide et $\phi : E \to E$ uniformément continue. Pour $q\geq 1$ entier, 
on note $\phi^q = \phi \circ \cdots \circ \phi$, $q$~fois. On suppose
qu'il existe $p \geq 1$ et $k \in ]0,1[$ tels que 
pour tout $x$, $y$ de $E$, on ait
\[
\min_{1 \leq q \leq p} d(\phi^q(x),\phi^q(y)) \leq k d(x,y) 
\]

\begin{questions}
 \item Soit $a=\inf_{x \in E} d(x,\phi(x))$. Montrer par l'absurde que $a=0$.
 \item Pour $t>0$ suffisamment petit, on définit le module de continuité de $\phi$ par
 \[
 \omega^1(t) = \sup_{x,y \in E, d(x,y) \leq t} d(\phi(x),\phi(y)) 
 \]
Montrer que $\omega^1(t)$ tend vers 0 quand $t$ tend vers 0.
 \item Pour $n>1$, montrer que $\phi^n$ est uniformément continue. On note $\omega^n(t)$ le module de continuité de $\phi^n$. 
 \item Pour $\alpha \in \RR^{+ \star}$, on pose $A_\alpha=\{x \in E / d(x,\phi(x)) \leq \alpha \}$.
% %  \begin{description}
% %  \item i) Montrer qu'il existe $\alpha$ tel que $A_\alpha$ soit non vide
% %  \item ii) En déduire que $A_{k^r\alpha}$ pour $r$ entier non nul est non vide
% %  \item iii) En déduire que pour tout $n \geq 1$ entier, $A_{1/n}$ est non vide.
% %  \item iv) Montrer que $A_{1/n}$ est une suite décroissante de parties fermées non vides de $E$.
% %  \end{description}
Montrer que pour tout $n \geq 1$ entier, $A_{1/n}$ est une suite décroissante de parties fermées non vides de $E$.
 \item Soient $x$, $y$ de $E$ et $q \in [1,p]$ tels que 
 \[
 d(\phi^q(x),\phi^q(y)) \leq k d(x,y) 
\]
Montrer en posant $ \phi^0(x)=x$ que 
\[
d(x,y) \leq \sum_{i=0}^{q-1} d(\phi^i(x),\phi^{i+1}(x)) + d(\phi^q(x),\phi^q(y)) + \sum_{i=0}^{q-1} d(\phi^i(y),\phi^{i+1}(y))
\]
\item On pose $\omega^0(t)=t$, en déduire que le diamètre de $A_{1/n}$ vérifie
\[
\delta(A_{1/n}) \leq \frac{2}{1-k}  \sum_{i=0}^{p-1} \omega^i(1/n)\,.
\]
\item  Que peut-on conclure sur $\phi$ ? 
\end{questions}
\end{exo}

\begin{exo}[Version faible du théorème de Picard et applications] 
\begin{questions}
\item Soient $(X,d)$ un espace métrique complet non vide et $f : X \rightarrow X$ une application (qu'on ne suppose pas continue).
On suppose qu'il existe $N \in \NN$ et $\alpha \in [0,1[$ tels que \\
$d(f^N(x),f^N(y)) \leq \alpha d(x,y)$  pour tous $x,y \in X$, c'est à dire que $f^N$ est une contraction. 
Montrer que $f$ a un unique point fixe $x_0$ et que, pour tout $x \in X$, la suite définie par $u_0 =x$ et $u_{n+1} = f(u_n)$ converge vers $x_0$.
Quelle est la vitesse de convergence de la suite $(u_n)_{n\in\NN}$ ?

\vskip1ex {\bf Applications.} 
Soient $a,b \in \RR$ et $I = [a,b]$ un intervalle compact. On considère une fonction $K : I \times I \ra \RR$ une fonction continue
et $\ph$ une fonction continue de $I$ dans $\RR$.
\item  On suppose que $(b-a)\norm{K}_{\infty} < 1$. Montrer qu'il existe une unique application continue $x : I \ra \RR$ telle que
\centersright[1]{$x(t) = \ph(t) + \int_{a}^{b} K(s,t)x(s) \d\,s\,.$}
\ni 

\item Montrer qu'il existe une unique fonction $x : I \ra \RR$ telle que 
\centersright[1]{$x(t) = \ph(t) + \int_{a}^{t} K(s,t)x(s) \d\,s\,.$}
\ni
\end{questions}
\end{exo}

\begin{exo}[Une démonstration plus topologique du théorème de Picard] \\
\textbf{1.} On notera, pour $A$ une partie d'un espace métrique, $\delta(A)=\sup \{d(x,y) \ / \ x,y \in A \}$ le diamètre de $A$. Démontrer le résultat suivant (théorème dit des fermés emboîtés) : \\
Un espace métrique $(X,d)$ est complet ssi l'intersection de toute suite décroissante $(A_{n})_{n \in \NN}$ de parties fermées non vides de $X$ telles que $\lim_{n\rightarrow + \infty} \delta(A_{n})=0$ est non vide. \vskip1ex
\noindent \textbf{2.} Soit $(X,d)$ un espace métrique complet non vide et $f : \ X\rightarrow X$ lipschitzienne de rapport $k \in [0,1[$. Pour $R \in \RR^{+}$, on pose $A_{R}=\{x \in X \ / \ d(x,f(x))\leq R\}$. 
\begin{questions}
\item Montrer que $f(A_{R}) \subset A_{kR}$ et en déduire que pour tout $R \in \RR^{+}_{\ast}$, $A_{R}$ est une partie fermée non vide de $X$.

\item Soient $x,y \in A_{R}$. Montrer que $d(x,y)\leq 2R+d \left ( f(x),f(y) \right )$ et en déduire que $\delta(A_{R})\leq \frac{2R}{1-k}$. 

\item Montrer que $A_{0}=\bigcap_{n \in \NN^{\ast}} A_{1/n}$ et en conclure que $A_{0}$ est non vide.
\end{questions}
\end{exo}


\begin{exo}[Suite de points fixes] Soient $(X,d)$ un espace métrique complet et $f_n : X \ra X$ une suite de fonctions continues.
On suppose que $f_n$ admet un point fixe $x_n$.

Pour les questions~\question{a}, \question{b} et~\question{c}, on suppose que $(f_n)_{n \in \NN}$ converge uniformément vers $f$ sur $X$.
\begin{questions}
\item On suppose que $(x_n)_{n \in \NN}$ converge (vers $x_0$). Montrer que $x_0$ est un point fixe pour $f$.

\item On suppose que $(f(x_n))_{n \in \NN}$ converge (vers $x_0$). Montrer que $x_0$ est un point fixe pour $f$.

\item On suppose que $f$ est contractante. Montrer que $(x_n)_{n \in \NN}$ converge vers l'unique point fixe de $f$. 

Pour les questions~\question{d}, \question{e} et~\question{f}, on suppose que $(f_n)_{n \in \NN}$ converge simplement vers $f$ sur $X$.
On suppose aussi qu'il existe $\alpha >0$, pour tout $n \in \NN$, $f_n$ soit $\alpha$-lipschiztienne.

\item Montrer que $f$ est $\alpha$-lipschiztienne. En déduire que si $(x_n)_{n \in \NN}$ converge vers $x_0$ alors $x_0$ est un point fixe de $f$.

\item On suppose $\alpha < 1$. Montrer que $(x_n)_{n\in \NN}$ converge vers l'unique point fixe de $f$. Montrer qu'on ne peut pas remplacer la condition $\alpha < 1$ par la condition $\alpha_n < 1$ pour tout $n \in \NN$ (où $f_n$ est $\alpha_n$-lipschitzienne).
On pourra considérer l'opérateur $f_n : \ell^2 \ra \ell^2$ défini par 
\centers[1]{$f_n((x_k)_{k \in \NN}) = (0,\ldots, 0, (1-1/n)x_n + 1/n,0,\ldots)$.}

\item {\bf Application.} Soient $X$ un compact non vide d'un espace vectoriel normé qu'on suppose étoilé par rapport à l'un de ses points $x_0$ 
(c'est le cas par exemple si $X$ est convexe). On suppose que $\norm{f(x)-f(y)} \leq \norm{x-y}$ pour tous $x,y \in X$. 
Montrer que $f$ a un point fixe (on pourra introduire une suite $(t_n)_{n \in \NN}$ qui tend vers $0$ et les fonctions $f_n(x) = (1-t_n)f(x) + t_n x_0$).
Peut-on retirer l'hypothèse de compacité ? (Pensez à une translation sur $\RR$).
\end{questions}
\end{exo}

\begin{exo}[Inversion globale et point fixe] 
\begin{questions} 
\item Soit $(Y,d)$ un espace métrique complet et $B= B(y_0,r)$ la boule ouverte de centre $y_0$ et de rayon $r$. 
On considère $f: B \ra Y$ une application contractante de rapport $\al <1$. Montrer que si $d(f(y_0),y_0) < (1-\al)r$ alors $f$ a un point fixe 
(on pourra introduire la boule fermé de centre $y_0$ et de rayon $\ep< r$ avec $\ep$ bien choisi qui est alors un espace complet). 

\vskip2ex Dans les questions qui suivent, on considère $X$ un espace de Banach, $U$ un ouvert de $X$ et $F : U \ra X$ une application contractante (de constante $\al$). 
Pour $x \in U$, on pose $f(x) = x - F(x)$. 

\item Soient $x \in U$ et $r > 0$ tels que $B(x,r) \subset U$. Montrer que $B(f(x), (1-\al)r) \subset f(B(x,r))$ (pour $y \in B(f(x), (1-\al)r)$, on pourra introduire
la fonction $G: u \mapsto F(u) + y$ et utiliser la question~\question{a}). 

\item Montrer que si $V$ est un ouvert de $U$ alors $f(V)$ est ouvert. 

\item Montrer que $f$ est un homéomorphisme de $U$ sur $f(U)$. 

\item On suppose que $U = X$. Montrer que $f(U) = X$ et que $f$ est un homéomorphisme de $X$ sur lui-même.  
\end{questions}
\end{exo}
