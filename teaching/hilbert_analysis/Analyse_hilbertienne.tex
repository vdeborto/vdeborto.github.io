\section{Généralités}

\subsection{Rappels sur les espaces préhilbertiens, \cite{gourdonan}}

\begin{notations}
Dans toute cette partie, $E$ désignera un espace vectoriel réel ou complexe muni d'un produit scalaire $\left\langle \cdot\,,\cdot \right\rangle$.
\end{notations}

\begin{lemma}\textbf{Identité du parallélogramme.}
$\forall x,y\in E,$
\begin{equation*}
\left\Vert x+y\right\Vert^2 + \left\Vert x-y\right\Vert^2 = 2\left(\left\Vert x\right\Vert^2 + \left\Vert y\right\Vert^2 \right).
\end{equation*} 
\end{lemma}

\begin{rem}
  On peut montrer que toute norme qui vérifie cette identité provient d'un produit scalaire (\cite{gourdonal} p.252.). De même mettre en défaut l'identité du parallélogramme permet de
  montrer que de nombreux espaces de Banach ne sont pas des espaces préhilbertiens. 
\end{rem}

\begin{ex}
  $L^1([0,1])$ n'est pas un espace de préhilbertien (considérer $x \mapsto 1$ et $x \mapsto 2x$ par exemple).
\end{ex}
  
\begin{prop}\textbf{Projection sur un convexe fermé.}
Soit $C$ un convexe complet non vide de $E$. 
\begin{enumerate}
\item Pour tout $x\in E$, il existe un unique élément de $C$, noté $P_C(x)$, tel que
\begin{equation*}
\left\Vert x-P_C(x) \right\Vert = \inf\limits_{c\in C} \left\Vert x-c \right\Vert = d(x,C).
\end{equation*}
\item $P_C(x)$ est caractérisé comme étant l'unique élément de $C$ tel que 
\begin{equation*}
\forall c\in C,\quad \left\langle x-P_C(x),c-P_C(x) \right\rangle \leq 0.
\end{equation*}
\item L'opérateur $P_C$ est 1-lipschitzien.
\item Si $C$ est un sous espace vectoriel fermé de $E$, $P_C(x)$ est caractérisé comme étant l'unique élément de $C$ tel que 
\begin{equation*}
\forall c\in C,\quad \left\langle x-P_C(x),c\right\rangle = 0
\end{equation*}
et l'opérateur $P_C$ est linéaire.
\end{enumerate}
\end{prop}

\begin{exo}\textbf{Contre-exemple dans un espace de Banach.}
  On considère
  \begin{equation*}
    \ell^{\infty}_0 = \left\lbrace u \in \mathbb{C}^{\mathbb{N}}, \ \underset{n \in \mathbb{N}} \sup \vert u_n \vert < +\infty,  \ \underset{n \rightarrow +\infty}{\lim} u_n = 0 \right\rbrace,
  \end{equation*}
  qui est un espace de Banach pour la norme infinie.
  \begin{enumerate}
  \item Montrer $H = \left\lbrace u \in \ell^{\infty}_0, \ \underset{n \in \mathbb{N}^*}{\sum} 2^{-n} u_n =1\right\rbrace$ est un fermé convexe de $\ell^{\infty}$.
  \item Montrer que la distance de $0$ à $H$ n'est pas atteinte.
    \item De même trouver un exemple d'espace de Banach, de convexe fermé et de point dans cet espace tel que la distance est atteinte en un nombre infini de points.
  \end{enumerate}
\end{exo}

\begin{rem}\textbf{Lien avec l'espérance conditionnelle, \cite{beck} p.100.}
  Soient $(X,A,\mu)$ un espace mesuré et $F$ une sous tribu de $A$ et $\nu$ la restriction de $\mu$ à $F$. Alors $L^2(X,F,\nu)$ est un sous espace fermé de $L^2(X,A,\mu)$, et on peut définir l'espérance conditionnelle sachant $F$ d'un élément de $L^2(X,A,\mu)$ comme son projeté orthogonal sur $L^2(X,F,\nu)$. Posons $(\Omega, A, P)$ un espace probabilisé, $X \in L^2(\Omega,A,P)$, $F = \sigma(X)$ et $\nu$ la restriction de $P$ à $F$. On appelle espérance conditionnelle sachant $X$ d'un élément de $L^2(\Omega,A,P)$ sur $L^2(\Omega,F,\nu)$. On note $E(\cdot | X)$ cette projection. La propriété de minimisation possède alors une interprétation en termes probabilistes :
  \begin{equation*}
    \forall (Y,Z) \in L^2(\Omega,A,P) \times L^2(\Omega,F,\nu), \ E((Y-Z)^2) \ge E((Y-E(Y | X))^2).
  \end{equation*}
  En se restreignant aux variables aléatoires de carré intégrable à moyenne nulle (qui est un espace
  de Hilbert, voir \ref{ex:hilbert_spaces}), on obtient que la variance de $Y - Z$ est minimale pour $Z=E(Y|X)$. \\
  Enfin il faut noter que l'espérance conditionnelle a été définie pour des variables aléatoires de carré intégrable. Une définition directe générale sur $L^1(\Omega,A,P)$ passe par le théorème de Radon-Nikodyn, \cite{rudin}. On peut néanmoins définir l'espérance conditionnelle sur $L^1(\Omega,A,P)$ par densité de $L^1(\Omega,A,P) \cap L^2(\Omega,A,P)$ dans $L^1(\Omega,A,P)$ et continuité en norme 1 de l'application qui à une variable aléatoire associe son espérance conditionnelle (voir TD de probabilité).
\end{rem}



\subsection{Rappels sur les espaces hilbertiens, \cite{gourdonan}}

\begin{definition}
On appelle espace hilbertien un espace préhilbertien qui est complet pour la norme issue du produit scalaire.
\end{definition}

\begin{ex}
\begin{enumerate}
\item Tout espace euclidien est un espace de Hilbert.
\item $\ds \ell^2=\left\{u\in\C^{\N},\ \sum_{n} \vert u_n\vert^2 < \infty\right\}$ est un espace de Hilbert.
\item $\ds L^2(X,A,\mu)=\left\{f \text{ mesurables sur }X,\ \int\limits_{X} \vert f\vert^2 d\mu(x) < \infty \right\}$ est un espace de Hilbert.
\item $\ds L^2_0([0,1])=\left\{f \text{ mesurables sur }[0,1],\ \int_0^1 \vert f\vert^2  < \infty,\ \int_0^1  f  =0 \right\}$ est un espace de Hilbert.
\end{enumerate}
\label{ex:hilbert_spaces}
\end{ex}

\begin{notations}
\begin{enumerate}
\item Dans toute la suite, $H$ désignera un espace hilbertien et $\LL(H)$ désignera l'ensemble des endomorphismes continus de $H$.
\item On note $H'$ le dual topologique de $H$, c'est à dire la restriction du dual algébrique $H^*$ aux formes linéaires continues.
\item Si $\varphi\in H'$ et $x\in H$, on utilisera parfois $\langle \varphi,x\rangle$ pour désigner $\varphi(x)$.
\end{enumerate}
\end{notations}

\begin{prop}
Soit $F$ un sous espace vectoriel de $H$.
\begin{enumerate}
\item $H=\overline F\oplus F^{\perp}$.
\item $F^{\perp\perp}=\overline F$.
\item $F$ est dense dans $H$ \ssi $F^{\perp}=\{0\}$.
\end{enumerate}
\end{prop}

\begin{exo}\textbf{Contre exemple.}
Soit $H=\ell^2$ et $F$ le sous espace de $H$ des suites nulles à partir d'un certain rang. Déterminer $F^{\perp}$, en déduire qu'on n'a pas $H=F\oplus F^{\perp}$.
\end{exo}


\begin{rem}
On peut étudier la série $\underset{n \in \mathbb{N}}{\sum}\frac{\epsilon_n}{n^{\alpha}}$ avec $\alpha > \frac{1}{2}$ et $(\epsilon_n)_{n \in \mathbb{N}}$ une suite de variables aléatoires indépendantes de Rademacher, i.e $P(\epsilon_n = 1) = P(\epsilon_n = -1) = \frac{1}{2}$. $S_n = \underset{k = 0}{\overset{n}{\sum}} \frac{\epsilon_k}{k^{\alpha}}$ est une martingale bornée dans $L^2(\Omega)$. En effet $E(S_n^2) = \underset{k=0}{\overset{n}{\sum}} \frac{1}{k^{2\alpha}}$, qui converge. Une connaissance de la théorie des martingales assure qu'elle converge presque-sûrement. On peut redémontrer ce résultat à la main en utilisant l'inégalité de Hoeffding et le lemme de Borel-Cantelli.
\end{rem}



\subsection{Théorème de représentation de Riesz-Fréchet, \cite{gourdonan}}

\begin{theorem}\textbf{Théorème de représentation de Riesz-Fréchet.}
Soit $\varphi\in H'$. Il existe un unique $y\in H$ tel que
\begin{equation*}
\forall x\in H,\quad \langle \varphi,x\rangle=\langle x,y\rangle.
\end{equation*}
De plus, $\left\Vert\varphi\right\Vert_{H'}=\left\Vert y\right\Vert$.
\end{theorem}

\begin{rem}
En d'autres termes, $y\mapsto \langle\cdot,y\rangle$ est une isométrie surjective de $H$ vers $H'$.
\end{rem}

\begin{cor}
$H$ est réflexif, c'est à dire que
\begin{equation*}
J : \left\{
\begin{aligned}
&H \to H'' \\
&x \mapsto [\varphi \mapsto \langle \varphi,x\rangle]
\end{aligned}
\right.
\end{equation*}
est surjective.%
\end{cor}
\begin{rem}
  Soit une norme $\|\cdot\|$ sur $L^1([0,1])$ tel que $\left(L^1([0,1]),\| \cdot \| \right)$ est
  topologiquement équivalent à $\left(L^1([0,1]),\|\cdot\|_1 \right)$ alors $\left(L^1([0,1]),\| \cdot \| \right)$ ne peut pas être un espace de Hilbert.
\end{rem}

\begin{prop}\textbf{Existence de l'adjoint.}
Pour tout $T\in \LL(H)$ il existe un unique $T^*\in \LL(H)$ tel que 
\begin{equation*}
\forall x,y\in H,\quad \langle Tx,y\rangle = \langle x,T^*y\rangle.
\end{equation*}
De plus $\Vert T\Vert = \Vert T^*\Vert$.
\end{prop}

\begin{rem}
  Le théorème d'Hellinger-Toeplitz (\cite{gt_af}, p.112) assure que deux applications linéaires $T$ et $U$ (non nécessairement continues) qui vérifient :
  \begin{equation*}
    \forall (x,y) \in H, \ \langle Tx,y \rangle = \langle x,Uy \rangle
  \end{equation*}
  sont des éléments de $\LL(H)$ (c'est une application du théorème du graphe fermé).
\end{rem}

\begin{exo}\textbf{$\blacktriangleright$ DEV $\blacktriangleleft$ Théorème ergodique de Von Neumann, \cite{beck} p.137.}
Soit $T\in \LL(H)$ tel que $\Vert T\Vert \leq 1$. On considère
\begin{equation*}
T_n=\frac{1}{n+1}\sum_{k=0}^n T^k.
\end{equation*}
On veut montrer que pour tout $x\in H$, $T_n(x)\to p(x)$, où $p$ est le projecteur orthogonal sur $\ker(I-T)$.
\begin{enumerate}
\item Montrer que $H=\ker(I-T^*)\oplus\overline{\im(I-T)}$.
\item Montrer que $\ker(I-T^*)=\ker(I-T)$.
\item Étudier séparément la limite de $T_n(x)$ pour $x$ dans $\ker(I-T)$ et $\im(I-T)$ puis conclure.
\item Exemple : on considère $H=L^2(\T)$, $f\in H$ et $\alpha\in\R\setminus 2\pi\Q$. Montrer que 
\begin{equation*}
\frac{1}{n}\sum_{k=0}^{n-1}f(\cdot+k\alpha) \underset{L^2}{\longrightarrow} \frac{1}{2\pi}\int_0^{2\pi} f.
\end{equation*}
\end{enumerate}
\end{exo}


\begin{exo}\textbf{Démonstration simplifiée des théorèmes de Hahn-Banach, \cite{beck} p.97 \& p.106.}
Démontrer, indépendamment et sans utiliser le lemme de Zorn, les théorèmes de Hahn-Banach géométriques et analytiques dans le cadre d'un espace de Hilbert.
\end{exo}

\section{Théorèmes de Stampacchia et de Lax-Milgram, vers les espaces de Sobolev et les solutions faibles, \cite{brezis}}

\subsection{Cadre théorique}

\begin{theorem}\textbf{Théorème de Stampacchia, \cite{brezis} p.83.}
Soit $a$ une forme bilinéaire continue et coercive sur $H$, i.e. il existe $\alpha>0$ tel que 
\begin{equation*}
\forall v\in H,\quad a(v,v)\geq \alpha \Vert v\Vert^2.
\end{equation*}
Soient $K$ un convexe fermé non vide de $H$ et $\varphi\in H'$. Alors il existe un unique $u\in K$ tel que 
\begin{equation*}
\forall v\in K,\quad a(u,v-u) \geq \langle\varphi,v-u\rangle.
\end{equation*}
Si de plus $a$ est symétrique, alors $u$ est caractérisé comme étant l'unique élément de $K$ tel que 
\begin{equation*}
\frac{1}{2}a(u,u)-\langle\varphi,u\rangle = \min\limits_{v\in K} \left(\frac{1}{2}a(v,v)-\langle\varphi,v\rangle\right).
\end{equation*}
\end{theorem}

\begin{exo}\textbf{$\blacktriangleright$ DEV $\blacktriangleleft$ Preuve.}
\begin{enumerate}
\item En utilisant le théorème de Riesz, interpréter $u$ comme un projeté orthogonal.
\item En remarquant qu'on a en fait un problème de point fixe, prouver qu'un opérateur bien choisi est contractant et en déduire l'existence et l'unicité de $u$. 
\item Dans le cas où $a$ est symétrique, considérer $H$ muni du produit scalaire défini par $a$ et reprendre la démarche de la première question.
\end{enumerate}
\end{exo}

\begin{theorem}\textbf{Théorème de Lax-Milgram, \cite{brezis} p.84.}
Soient $a$ une forme bilinéaire continue et coercive sur $H$ et $\varphi\in H'$. Alors il existe un unique $u\in H$ tel que 
\begin{equation*}
\forall v\in h,\quad a(u,v) = \langle\varphi,v\rangle.
\end{equation*}
Si de plus $a$ est symétrique, alors $u$ est caractérisé comme étant l'unique élément de $H$ tel que 
\begin{equation*}
\frac{1}{2}a(u,u)-\langle\varphi,u\rangle = \min\limits_{v\in H} \left(\frac{1}{2}a(v,v)-\langle\varphi,v\rangle\right).
\end{equation*}
\end{theorem}

\begin{rem}
\begin{enumerate}
\item Le théorème de Lax-Milgram peut-être vu comme un corollaire du théorème de Stampacchia, mais il peut aussi se démontrer plus simplement, directement à partir du théorème de Riesz (\cite{hirla} p.101.).
\item Une formulation équivalente du théorème de Lax-Milgram est la suivante :\\
soit $L:H\to H'$ linéaire continue et telle que 
\begin{equation*}
\exists \alpha>0,\ \forall u\in H,\quad \langle Lu,u\rangle \geq \alpha \left\Vert u \right\Vert^2,
\end{equation*}
alors $L$ est un isomorphisme de $H$ vers $H'$ \textit{(exercice : vérifier cette équivalence)}.
\item Le théorème de Lax-Milgram est un outil très utile pour l'étude des EDP elliptiques linéaires, voir \cite{brezis} pour de multiples exemples.
\end{enumerate}
\end{rem}

\subsection{Une application : l'espace $H^1_0$ et la résolution d'un problème elliptique avec conditions aux bords de Dirichlet en 1D, \cite{brezis}}

\begin{definition}\textbf{Espace $H^1$.}
On note $I=]0,1[$. On définit
\begin{equation*}
H^1(I) = \left\{u\in L^2(I),\ u'\in L^2(I)\right\}.
\end{equation*} 
\end{definition}

\begin{rem}
On rappelle que pour $u\in L^2(I)$, $u'\in L^2(I)$ signifie qu'il existe $v\in L^2(I)$ tel que
\begin{equation*}
\forall \varphi \in \CC^{\infty}_c(I),\quad \int\limits_I u\varphi' = -\int\limits_I v\varphi.
\end{equation*}
Dans ce cas, on dit que $v$ est la dérivée faible de $u$ (elle est forcément unique), et on note $u'=v$. On notera que si $u$ est de classe $\CC^1$, sa dérivée faible coïncide avec sa dérivée classique.
\end{rem}

\begin{prop}\textbf{\cite{brezis} p.121.}
$H^1(I)$ muni du produit scalaire
\begin{equation*}
\langle u,v\rangle = \int\limits_I\left(uv + u'v'\right)
\end{equation*}
est un espace de Hilbert séparable.
\end{prop}

\begin{lemma}\textbf{\cite{brezis} p.123.}
Soient $g\in L^1(I)$ et $y_0\in I$. Alors la fonction $v$ définie par
\begin{equation*}
v(x)=\int_{y_0}^x g(t)dt
\end{equation*}
est continue sur $I$ et vérifie
\begin{equation*}
\int\limits_I v\varphi' = -\int\limits_I g\varphi,\quad \forall \varphi\in \CC^{\infty}_c(I).
\end{equation*}
En particulier, si $g\in L^2(I)$, alors $v\in H^1(I)$ et $v'=g$.
\end{lemma}

\begin{lemma}\textbf{\cite{brezis} p.122.}
Soit $f\in L^1(I)$ telle que
\begin{equation*}
\int_I f\varphi'=0 \quad \forall \varphi\in \CC^{\infty}_c(I),
\end{equation*}
alors il existe une constante $C$ telle que $f=C$ presque partout.
\end{lemma}

\begin{theorem}\textbf{Un cas particulier d'injection de Sobolev, \cite{brezis} p.122.}
Soit $u\in H^1(I)$. Il existe un représentant $\tilde u$ de $u$ qui est continu sur $\overline I$ et vérifie
\begin{equation*}
\tilde  u(y) - \tilde u(x) = \int_x^y u'(t)dt, \quad \forall x,y\in \overline I.
\end{equation*}
\end{theorem}

\begin{definition}
On définit 
\begin{equation*}
H^1_0(I) = \left\{ u\in H^1(I), u(0)=0=u(1) \right\}.
\end{equation*}
\end{definition}

\begin{theorem}\textbf{Un cas particulier de l'inégalité de Poincaré, \cite{brezis} p.134.}
Il existe une constante $C$ telle que 
\begin{equation*}
\forall u\in H^1_0(I),\quad \left\Vert u \right\Vert_{H^1(I)} \leq C \left\Vert u' \right\Vert_{L^2(I)}.
\end{equation*}
On en déduit que $\left\Vert u' \right\Vert_{L^2(I)}$ est une norme sur $H^1_0(I)$ équivalente à $\ds \left\Vert u \right\Vert_{H^1(I)}$.
\end{theorem}

\begin{theorem}\textbf{Un résultat de densité.}
$\CC^{\infty}_c(I)$ est dense dans $H^1_0(I)$.
\end{theorem}

\begin{rem}
Traditionnellement, les espaces de type $H^1_0$ sont définis comme la fermeture de $\CC^{\infty}_c$ dans $H^1$, et on démontre ensuite que cela correspond aux fonctions nulles au bord (en un sens à préciser en dimension $>1$).
\end{rem}

\begin{definition}\textbf{Solution faible, \cite{brezis} p.136.}
\label{def:sol_faible}
Soit $f\in L^2(I)$. On cherche à trouver $u$ solution du problème de Dirichlet homogène :
\begin{equation}\label{eq:dirichlet_homo}
\left\{
\begin{aligned}
&-u''+u=f \quad \text{sur }I \\
&u(0)=0=u(1).
\end{aligned}
\right.
\end{equation}
\begin{enumerate}
\item Une solution forte de \eqref{eq:dirichlet_homo} est une fonction $u\in \CC^2(I)\cap \CC(\overline I)$ où les dérivées sont à comprendre au sens classique.
\item Une solution faible de \eqref{eq:dirichlet_homo} est une fonction $u\in H^1_0(I)$ qui vérifie
\begin{equation*}
\int\limits_I\left(u'v' + uv\right) = \int\limits_I fv,\quad \forall v\in H^1_0(I).
\end{equation*}
\end{enumerate}
\end{definition}

\begin{rem}
Toute solution forte de \eqref{eq:dirichlet_homo} est une solution faible de \eqref{eq:dirichlet_homo}.
\end{rem}

\begin{theorem}\textbf{\cite{brezis} p.136.}
Soit $f\in L^2(I)$. Il existe un unique $u\in H^1_0(I)$ solution faible de \eqref{eq:dirichlet_homo}. De plus, $u$ est caractérisé comme étant l'unique solution au problème de minimisation :
\begin{equation*}
\min\limits_{v\in H^1_0(I)} \frac{1}{2}\int\limits_I\left(v^2+v'^2\right) - \int\limits_I fv. 
\end{equation*} 
Enfin, si $f\in \CC(\overline{I})$ alors $u$ est une solution forte de \eqref{eq:dirichlet_homo}.
\end{theorem}

\begin{exo}\textbf{Preuve.}
Démontrer le théorème précédent en utilisant le théorème de Lax-Milgram. Pour montrer que $u$ est une solution forte, on pourra d'abord montrer que $u$ appartient à l'espace $H^2(I)=\{u\in L^2(I),\ u',u''\in L^2(I)\}$.
\end{exo}

\begin{exo}\textbf{Condition aux bords VS espace de fonction.}
Soit $u\in H^1(I)$ telle que
\begin{equation*}
\int\limits_I \left(u'v'+uv\right)=\int\limits_I fv,\quad \forall v\in H^1(I).
\end{equation*}
Montrer que $u$ est solution d'un problème similaire à~\eqref{eq:dirichlet_homo} (qu'on appelle problème de Neumann homogène).
\begin{rem}Pour introduire les conditions de Dirichlet on a modifié l'espace de départ ($H^1_0(I)$ au lieu de $H^1(I)$), par contre pour introduire les conditions de Neumann c'est la forme linéaire qui est modifiée. Les conditions de Dirichlet sont dites \textbf{essentielles}, ou explicites, les conditions de Neumann sont dites \textbf{naturelles}, ou implicites.
\end{rem}
\end{exo}
\begin{rem}
  Dans les exercices précédents on a utilisé le théorème de Lax-Milgram version symétrique qui n'est
  autre que le théorème de Riesz-Fréchet. Un exemple d'application du théorème de Lax-Milgram version
  non symétrique est donné par le problème suivant (\cite{brezis} p.138) :
  \begin{equation*}
    \left\{
      \begin{aligned}
        &-(pu')'+ru'+qu = f \quad \text{sur }I\\
        &u(0)=u(1)=0 \\
      \end{aligned}
      \right.
    \end{equation*}
    Noter qu'il n'y a pas automatiquement de problème de minimisation associée à cette équation.
\end{rem}
\begin{theorem}\textbf{\cite{brezis} p.137.}
Soient $\alpha,\beta \in\R$ et $f\in L^2(I)$. Il existe un unique $u\in H^2(I)$ solution du problème de Dirichlet non homogène : 
\begin{equation}\label{eq:dirichlet_non_homo}
\left\{
\begin{aligned}
&-u''+u=f \quad \text{sur }I \\
&u(0)=\alpha \text{ et } u(1)=\beta.
\end{aligned}
\right.
\end{equation} 
De plus, $u$ est caractérisé comme étant l'unique solution au problème de minimisation :
\begin{equation*}
\min\limits_{\substack{v\in H^1(I) \\ v(0)=\alpha,\ v(1)=\beta}} \frac{1}{2}\int\limits_I\left(v^2+v'^2\right) - \int\limits_I fv. 
\end{equation*} 
Enfin, si $f\in \CC(\overline{I})$ alors $u$ est une solution forte de \eqref{eq:dirichlet_non_homo}.
\end{theorem}

\begin{exo}\textbf{Preuve.}
Démontrer le théorème précédent en utilisant le théorème de Stampacchia. \textit{On pourra considérer le convexe fermé de $H^1(I)$ défini par $K=\{v\in H^1(I), v(0)=\alpha,\ v(1)=\beta\}$.}
\end{exo}

\begin{rem}
On aurait aussi pu démontrer ce résultat en se ramenant au problème de Dirichlet homogène.
\end{rem}

\begin{exo}\textbf{Résolution \emph{à la main}, méthode de tir.} Soit $f\in\CC(I)$. Pour tout $\gamma\in\R$, on note $u_{\gamma}$ la solution de
\begin{equation*}
\left\{
\begin{aligned}
&-u''+u=f \quad \text{sur }I \\
&u(0)=\alpha \text{ et } u'(0)=\gamma.
\end{aligned}
\right.
\end{equation*} 
En considérant $u_{\gamma}-u_1$, avec un $\gamma$ judicieusement choisi, montrer que~\eqref{eq:dirichlet_non_homo} admet une unique solution.
\end{exo}

\begin{exo}\textbf{Résolution explicite avec $f \in L^2(I)$}
  Soit $f = \chi_{[\frac{1}{2},1[}$.
  \begin{enumerate}
  \item Montrer $-u''+u=f$ avec conditions aux bords de Dirichlet homogène admet une unique solution dans $H^1_0(I)$.
  \item Montrer que $u$ est solution forte $-u''+u=f$ sur $]0,\frac{1}{2}[$ et sur $]\frac{1}{2},1[$.
    \item Expliciter la solution faible.
  \end{enumerate}
\end{exo}

\section{Topologie dans un Hilbert, \cite{gourdonan} p.411.}

\begin{definition}\textbf{Convergence faible.}
Une suite $(u_n)\in H^{\N}$ converge faiblement vers $u\in H$ si
\begin{equation*}
\forall v\in H,\quad \langle u_n,v\rangle \to \langle u,v\rangle.
\end{equation*}
\end{definition}

\begin{rem}
Cette définition coïncide bien avec celle vue en analyse fonctionnelle puisque d'après le théorème de représentation de Riesz, toute forme linéaire continue sur $H$ peut se représenter comme un produit scalaire par un certain vecteur de $H$.
\end{rem}

\begin{exo}
\begin{enumerate}
\item Donner un exemple de suite qui converge faiblement mais pas fortement.
\item Donner un exemple d'une suite $(u_n)\in \left(L^2([0,1])\right)^{\N}$ qui converge faiblement vers $u$ et telle que $u_n^2\in L^2([0,1])$ pour tout $n\in\N$, mais pour laquelle $(u_n^2)$ ne converge pas faiblement vers $u^2$. 
\item Montrer que toute suite qui converge faiblement est bornée \textit{(on pourra utiliser le théorème de Banach-Steinhaus)} et que la limite faible $u$ vérifie $\Vert u \Vert \leq \liminf \Vert u_n\Vert$.
\item Soit $(u_n)$ une suite qui converge faiblement vers $u$ et telle que $\Vert u_n\Vert \to \Vert u\Vert$. Montrer que $(u_n)$ converge fortement vers $u$.
\item Soient $(u_n)$ une suite qui converge faiblement vers $u$ et $(v_n)$ une suite qui converge fortement vers $v$. Montrer que $\langle u_n,v_n \rangle$ converge vers $\langle u,v\rangle$.
\item Montrer que de toute suite bornée $(u_n)$, on peut extraire une sous suite faiblement convergente \textit{(on pourra utiliser le procédé d'extraction diagonale)}.
\end{enumerate}
\end{exo}


\begin{rem}
La dernière question de l'exercice précédent montre que la boule unité de $H$ est séquentiellement compacte pour la topologie faible. On a donc redémontré, dans le cas particulier des espaces de Hilbert, la version \emph{utile} du théorème de Banach-Alaoglu (mais pas la version générale, car on n'a pas supposé que $H$ était séparable, la boule unité fermée n'est donc pas forcément métrisable pour la topologie faible et on n'a donc pas \emph{compacité séquentielle} $\Rightarrow$ \emph{compacité}). Ce résultat de compacité séquentielle faible permet notamment de prouver de manière alternative (sans utiliser le théorème de Lax-Milgram) l'existence de solutions faibles pour \eqref{eq:dirichlet_homo} par exemple. C'est l'objet de l'exercice suivant.
\end{rem}

\begin{exo}\textbf{Approche variationnelle, \cite{allaire}.}
Dans le but de trouver une solution faible de \eqref{eq:dirichlet_homo}, on introduit la fonctionnelle $F$ définie sur $H^1_0(I)$ par 
\begin{equation*}
F(v)=\frac{1}{2}\int\limits_I\left(v^2+v'^2\right) - \int\limits_I fv.
\end{equation*}
\begin{enumerate}
\item Montrer que $\inf\limits_{v\in H^1_0(I)} F(v)$ est fini. On considère maintenant une suite minimisante $(u_n)\in H^1_0(I)^{\N}$, i.e. telle que $\lim F(u_n) = \inf\limits_{H^1_0(I)} F$.
\item Justifier que $(u_n)$ est bornée et qu'il existe $u\in H^1_0(I)$ tel que, quitte à extraire, $u'_n\underset{L^2}{\rightharpoonup} u'$ et $u_n\underset{L^2}{\rightharpoonup} u$.
\item En déduire que $F(u)=\inf\limits_{H^1_0(I)} F$.
\item En utilisant que pour tout $v\in H^1_0(I)$ et tout $\varepsilon\in\R$, $F(u)\leq F(u+\varepsilon v)$, montrer que $u$ est solution faible de \eqref{eq:dirichlet_homo}.
\item Montrer par un argument de convexité que cette solution est unique.
\end{enumerate}
\end{exo}

\begin{rem}
\begin{enumerate}
\item Si on avait voulu appliquer au problème \eqref{eq:dirichlet_non_homo} une approche similaire, on aurait eu besoin du lemme de Mazur qui dit que les convexes fermés pour la topologie forte le sont aussi pour la topologie faible (voir TD d'analyse fonctionnelle). \textit{Exercice : redémontrer ce résultat dans le cadre des espaces de Hilbert.}
\item Pour des problèmes linéaires, il est en général plus simple d'utiliser les théorèmes de Lax-Milgram ou de Stampacchia. L'approche variationnelle devient vraiment intéressante pour étudier des problèmes non linéaires, par exemple
\begin{equation*}
\left\{
\begin{aligned}
&-u''+u+u^3=f \quad \text{sur }I \\
&u(0)=0=u(1).
\end{aligned}
\right.
\end{equation*} 
On pourra alors étudier la fonctionnelle définie sur $H^1_0(I)$ par 
\begin{equation*}
F(v)=\frac{1}{2}\int\limits_I\left(v^2+v'^2\right) +\frac{1}{4}\int\limits_I v^4 - \int\limits_I fv,
\end{equation*}
la seule différence par rapport à l'exercice précédant étant qu'il faut un argument de compacité supplémentaire (voir proposition~\ref{prop:inj_compacte}) pour montrer que $\ds \int\limits_I u_n^4 \to \int\limits_I u^4$.
\end{enumerate}
\end{rem}

\section{Bases hilbertiennes}

\subsection{Généralités, \cite{gourdonan} p.408.}

\begin{prop}\textbf{Inégalité de Bessel.}
\label{prop:Bessel}
Soient $(e_n)_{n\in\N}$ une famille orthonormée de $H$ et $v\in H$. On note $\ds E=\overline{\Vect(e_n)_{n\in\N}}$. La série $\ds w=\sum_{n\in\N}\langle v,e_n\rangle e_n$ converge, $w$ est la projection orthogonale de $v$ sur $E$ et
\begin{equation*}
\Vert w\Vert^2 = \sum_{n\in\N} \vert\langle v,e_n\rangle\vert^2 \leq \Vert v\Vert^2 \quad \text{(inégalité de Bessel)}.
\end{equation*}
\end{prop}

\begin{definition}\textbf{Base hilbertienne}
Une base hilbertienne de $H$ est une famille orthonormée $(e_n)_{n\in\N}$ telle que $\ds E=\overline{\Vect(e_n)_{n\in\N}}=H$.
\end{definition}

\begin{prop}
\label{prop:base_hilbertienne}
Tout espace de Hilbert séparable admet une base hilbertienne et est isométriquement isomorphe à $\ell^2$. Si $(e_n)_{n\in\N}$ est une base hilbertienne de $H$, alors tout $v\in H$ s'écrit de manière unique comme
\begin{equation*}
v=\sum_{n\in\N}v_n e_n,
\end{equation*}
où $v_n=\langle v,e_n\rangle$, et on a
\begin{equation*}
\Vert v\Vert^2 = \sum_{n\in\N} \vert v_n\vert^2 \quad \text{(égalité de Parseval)}.
\end{equation*}
\end{prop}

\begin{exo}\textbf{Lien avec la convergence faible.}
Soit $\left(x^n\right)\in \left(\ell^2\right)^\N$. Montrer que
\begin{equation*}
x^n \rightharpoonup x \Longleftrightarrow \left\{
\begin{aligned}
&(x^n) \text{ est bornée}\\
&\forall p\in\mathbb{N},\ x^n_p\longrightarrow x_p.
\end{aligned}
\right.
\end{equation*}
\end{exo}


\subsection{Des exemples de bases hilbertiennes}
\subsubsection{Analyse de Fourier}
\begin{rem}
On rappelle (cf. TD de topologie), qu'un espace vectoriel normé admettant une base algébrique dénombrable n'est pas complet. En dimension infinie, une base hilbertienne n'est donc jamais une base algébrique!
\end{rem}

\begin{ex}\textbf{Séries de Fourier.}
$(t\mapsto e^{int})_{n\in\Z}$ est une base hilbertienne de $L^2(\T)$.
\end{ex}

\begin{rem}
  L'orthogonalité de ces fonctions est facile à montrer. Il reste à montrer que l'espace vectoriel
  qu'elles engendrent est dense dans les fonctions de carré intégrable sur le tore. On peut utiliser
  le théorème de Féjer (voir TD d'analyse de Fourier) ou le théorème de Stone-Weierstrass.
\end{rem}

On rappelle maintenant quelques résultats sur la transformée de Fourier (voir le TD d'analyse de Fourier) qui seront utiles pour traiter les exemples suivants.

\begin{definition}\textbf{Transformée de Fourier.}
Soit $f\in L^1(\R)$, on peut définir sa transformée de Fourier par 
\begin{equation*}
\hat f (\xi) = \int\limits_{\R} f(x) e^{-i x\xi} dx.
\end{equation*}
\end{definition}

\begin{theorem}\textbf{Formule d'inversion de Fourier.}
Si $f,\hat f \in L^1(\R)$, alors 
\begin{equation*}
f(x)=\frac{1}{2\pi}\int\limits_{\R} \hat f(\xi) e^{i x\xi} d\xi.
\end{equation*} 
\end{theorem}

\begin{theorem}\textbf{Fourier-Plancherel.}
  La transformée de Fourier s'étend en un isomorphisme involutif de $L^2(\R)$ dans lui même et vérifie
  \begin{equation*}
    \forall f \in L^2(\R), \ \| f \|_2^2 = \frac{1}{2\pi} \| \hat{f} \|_2^2
  \end{equation*}
\end{theorem}

\begin{exo}\textbf{$\blacktriangleright$ DEV $\blacktriangleleft$ Théorème d'échantillonnage de Shannon, \cite{wil} p.126.}
On définit 
\begin{equation*}
BL_2(\R)=\left\{ u\in L^2(\R),\ \supp(\hat u) \subset \left[-\pi,\pi\right] \right\}. 
\end{equation*}
\begin{enumerate}
\item Vérifier que $BL_2(\R)$ est un espace de Hilbert.
\item Soit $u\in BL_2(\R)$. Montrer que $u$ admet un représentant continu, tendant vers $0$ à l'infini et tel que 
\begin{equation*}
\left\Vert u \right\Vert_{L^{\infty}(\R)} \leq \left\Vert u \right\Vert_{L^{2}(\R)}.
\end{equation*}
\item On définit, pour $k\in \Z$, la fonction $e_k$ par $e_k(x)=e^{i kx}\mathbf{1}_{\left[-\pi,\pi\right]}(x)$. En considérant $\hat{e_k}$ déterminer une base hilbertienne de $BL_2(\R)$.
\item En déduire que pour tout $u\in BL_2(\R)$,
\begin{equation*}
u(x) = \sum_{k\in \Z}u(k)\sinc(x-k),
\end{equation*}
où $\ds \sinc(x)=\frac{\sin(\pi x)}{\pi x}$ et l'égalité vaut dans $L^2(\R)$.
\item Montrer que l'égalité de la question précédente est encore vraie lorsque l'on remplace la convergence dans $L^2(\R)$ par une convergence uniforme et que l'on remplace $u$ par son représentant continu.
\end{enumerate}
\end{exo}

\begin{exo}\textbf{$H^1(\T)$ et coefficients de Fourier}
  On définit
  \begin{equation*}H^1(\T) = \lbrace f \in L^2(\T), \exists g \in L^2(\T), \ \forall \varphi \in \mathcal{C}_c^{\infty}(\T), \ \int f\varphi' = -\int g\varphi \rbrace.
  \end{equation*}
  Soit $f\in H^1(\T)$.
  \begin{enumerate}
    \item Montrer que $H^1(\T) = \left\lbrace f \in H^1(]0,2\pi[), \ f(0) = f(2\pi) \right\rbrace$.
  \item Montrer que $\forall n \in \mathbb{Z}, \ c_n(f') =  inc_n(f)$.
    \item En déduire que $\forall x \in \T, \ f(x) = \underset{ n \in \mathbb{Z}}{\sum}c_n(f) e^{inx}$.
  \end{enumerate}
\end{exo}  

\begin{exo}\textbf{Ondelette de Haar, \cite{clf3} p.55 et \cite{peyre} p.54}
  Soit $\psi = \chi_{]0,1[}$ et $\phi = \chi_{]0,\frac{1}{2}[}-\chi_{]\frac{1}{2},1[}$. On définit
  \begin{equation*}
    \forall n \in \N, \forall k \in \Z, \ \forall x \in \R, \ \psi_{n,k}(x) = \psi(2^nx -k), \phi_{n,k}(x) = \phi(2^nx -k).
  \end{equation*}
  On note $V_N =\overline{\Vect \left( \psi_{n,k}, n \in \lbrace 0,\dots,N \rbrace, \ k \in \mathbb{Z} \right)}$.
  \begin{enumerate}
  \item Montrer $\overline{\underset{N \in \mathbb{N}}{\bigcup}{V_N}} = L^2(\R)$.
  \item Montrer que $(\phi_{N,k})_{k \in \mathbb{Z}}$ est une base hilbertienne de l'orthogonal à $V_N$ dans $V_{N+1}$.
    \item Donner une base hilbertienne de $L^2(\R)$.
  \end{enumerate}
\end{exo}

\subsubsection{Polynômes orthogonaux, \cite{pom} p.245}
Dans toute cette partie, $I$ est un intervalle de $\R$ d'intérieur non vide et $w:I\to\R$ une fonction poids, i.e. $w$ est mesurable, $w(x)>0$ pour presque tout $x$ et 
\begin{equation*}
\int\limits_I \vert x\vert^n w(x)< +\infty,\quad \forall n\in\N.
\end{equation*}
On considère
\begin{equation*}
L^2_w(I)=\left\{ f:I\to\C \text{ mesurable},\ \int\limits_I \vert f(x)\vert^2 w(x)dx <+\infty \right\},
\end{equation*}
muni du produit scalaire
\begin{equation*}
\langle f,g\rangle = \int\limits_I f(x)\overline{g(x)} w(x)dx,
\end{equation*}
qui fait de $L^2_w(I)$ un espace de Hilbert.

\begin{exo}\textbf{$\blacktriangleright$ DEV $\blacktriangleleft$ Densité des polynômes orthogonaux, \cite{beck} p.110 \& p.140.}
Le but de cet exercice est de construire une base hilbertienne particulière pour cet espace, sous certaines hypothèses sur $w$.
\begin{enumerate}
\item Rappeler brièvement pourquoi il est possible de définir une unique famille orthogonale $(P_n)_{n\in\N}$ de polynômes unitaires et tels que pour tout $n\in\N$, $\deg(P_n)=n$.
\item Si $I$ est borné, montrer que $(P_n)_{n\in\N}$ est une base hilbertienne de $L^2_w(I)$.
\item A partir de maintenant, $I$ n'est plus nécessairement borné mais on suppose qu'il existe $\alpha>0$ tel que 
\begin{equation*}
\int\limits_I e^{\alpha\vert x\vert}w(x)dx < +\infty.
\end{equation*}
Soit $f\in L^2_w(I)$. Montrer que la fonction $\varphi$ définie par
\begin{equation*}
\varphi(x) = 
\begin{cases}
f(x)w(x)\quad &\text{si }x\in I, \\
0 &\text{sinon},
\end{cases}
\end{equation*}
est bien dans $L^1(\R)$ et que sa transformée de Fourier se prolonge en une fonction $F$ holomorphe sur $\left\{z\in\C,\ \vert\Im(z)\vert<\frac{\alpha}{2} \right\}$.
\item Calculer $F^{(n)}(0)$ pour tout $n\in\N$ et en déduire que si $f\in \left(\R[X]\right)^{\perp}$ alors $f=0$. Conclure.
\item Montrer que si $I=]0,+\infty[$ et $w(x)=x^{-\ln x}$, les polynômes orthogonaux ne forment pas une base hilbertienne pour $L^2_w(I)$ (\textit{on pourra considérer la fonction définie par $f(x)=\sin(2\pi \ln x)$}).
\end{enumerate}
\end{exo}

\begin{exo}
Montrer que dans l'exercice précédent chaque polynôme $P_n$ admet $n$ racines réelles distinctes dans l'intérieur de l'intervalle $I$.
\end{exo}

\begin{prop}[Relation de récurrence]
Les polynômes $(P_n)_{n \in \mathbb{N}}$ vérifient une relation de récurrence linéaire d'ordre deux.
\end{prop}

\begin{ex}
  Quelques exemples de familles de polynômes :
  \begin{enumerate}
  \item Pour $I = [0,+\infty[$, $w: \ t  \mapsto  e^{-t}$. Les polynômes identifiés sont les
    polynômes de Laguerre
    \begin{equation*}
      \forall x \in \R, \forall n \in N, \ P_n(x) = \frac{e^x}{n!}\left( x \mapsto e^{-x}x^n\right)^{(n)}(x).
    \end{equation*}
    Ils vérifient la relation de récurrence suivante
    \begin{equation*}
      \forall x \in \R, \forall n \in \N, \ (n+1)P_{n+1}(x) = (2n+1-x)P_n(x) - nP_{n-1}(x).
    \end{equation*}
  \item Pour $I=[-1,1]$, $w: \ t  \mapsto  1$. Les polynômes identifiés sont les polynômes
    de Legendre
    \begin{equation*}
      \forall x \in \R, \forall n \in \N, \ P_n(x) = \frac{1}{2^n n!} \left( x \mapsto (x^2-1)^n \right)^{(n)}(x).
    \end{equation*}
  \end{enumerate}
\end{ex}
\section{Réduction des opérateurs compacts auto-adjoints}

\subsection{Cadre théorique, \cite{brezis,gourdonan}}

\begin{rem}
On considère ici pour simplifier $H$ un $\R$ espace de Hilbert. Une partie des résultats de cette section restent valable dans un espace de Banach quelconque, mais ils sont tous énoncés ici dans le cadre des espaces de Hilbert qui permet des preuves plus simples, voir \cite{brezis} pour le cadre général.
\end{rem}

\begin{definition}\textbf{Opérateur compact.}
Soit $T\in \LL(H)$. $T$ est dit compact si $T(\overline B(0,1))$ est relativement compact (i.e. $\overline{T(\overline B(0,1))}$ est compact).
\end{definition}

\begin{prop}
Soient $H_1$, $H_2$ deux espaces de Hilbert, $T_1:H\to H_1$ et $T_2:H_1\to H_2$ des applications linéaires continues. Si $T_1$ ou $T_2$ est compact, alors $T_2\circ T_1$ est compact.
\end{prop}

\begin{prop}\textbf{Caractérisation dans un Hilbert.}
Soit $T\in \LL(H)$. Montrer que $T$ est compact \ssi pour toute suite $(u_n)$ convergeant faiblement vers $u$, $Tu_n$ converge fortement vers $Tu$.
\end{prop}

\begin{rem}\textbf{Limite d'opérateurs de rang fini}
Dans un espace de Hilbert tout opérateur continu compact peut être vu comme limite (en terme de norme d'opérateur) d'opérateurs linéaires continus de rangs finis (\cite{schwartz}, p.425, \cite{brezis} p.89).
\end{rem}

\begin{exo}
  Soit $H$ un $\R$ espace de Hilbert séparable et $(e_n)_{n \in \mathbb{N}}$ une base hilbertienne
  de cet espace. Soit $T \in \LL(H)$ tel que $\forall n \in \N, \ T(e_n) = \lambda_n e_n$ avec $(\lambda_n)_{n \in \N} \in \R^{\N}$. Montrer que $T$ est compact \ssi $\lambda_n \ \underset{n \rightarrow +\infty}{\longrightarrow} 0$.
\end{exo}

\begin{exo}\textbf{Opérateur intégral}
  Soit $K \in \mathcal{C}([0,1] \times [0,1], \R)$. On définit $T: L^2([0,1]) \ \rightarrow L^2([0,1])$
  \begin{equation*}T(f)(x) = \int_0^1 K(x,y)f(y) \text{d}y . \end{equation*}
  \begin{enumerate}
  \item Montrer que $T$ est une application linéaire continue.
  \item Montrer que la restriction de $T$ aux fonctions continues à valeurs dans les fonctions continues est un opérateur linéaire continu compact.
  \item Montrer que $T$ est encore un opérateur linéaire continu et compact si $K \in L^2([0,1] \times [0,1])$ (\cite{clf3} p.65).
    \end{enumerate}
  \end{exo}

\begin{definition}\textbf{Spectre, valeurs propres.}
Soit $T\in \LL(H)$. On définit son spectre par
\begin{equation*}
\sigma(T) = \left\{ \lambda\in\R,\ T-\lambda I \text{ est non bijectif} \right\},
\end{equation*}
et l'ensemble de ses valeurs propres par
\begin{equation*}
\Lambda(T) = \left\{ \lambda\in\R,\ T-\lambda I \text{ est non injectif} \right\}.
\end{equation*}
\end{definition}

\begin{rem}
\begin{enumerate}
\item Si $\dim H = +\infty$, $\Lambda(T)$ peut très bien être strictement inclus dans $\sigma(T)$ (un endomorphisme peut être injectif mais non surjectif).
\item Cependant on peut montrer (mais on n'en aura pas besoin ici) que si $T$ est compact, alors $\sigma(T)\setminus\{0\} = \Lambda(T)\setminus \{0\}$.
\end{enumerate}
\end{rem}

\begin{prop}\textbf{Spectre d'un opérateur compact.}
\label{prop:spectre}
Soit $T\in \LL(H)$ compact.
\begin{enumerate}
\item Si $\lambda\in\Lambda(T)\setminus \{0\}$, alors $\ker(T-\lambda I)$ est de dimension finie.
\item $0$ est le seul point d'accumulation possible de $\Lambda(T)$.
\item $\Lambda(T)$ est fini ou dénombrable.
\end{enumerate}
\end{prop}

\begin{definition}\textbf{Opérateur auto-adjoint.}
Soit $T\in \LL(H)$. On dit que $T$ est auto-adjoint si $T^*=T$.
\end{definition}

\begin{lemma}\textbf{Norme d'un opérateur auto-adjoint.}
Soit $T\in \LL(H)$ auto-adjoint. Alors 
\begin{equation*}
\Vert T \Vert = \sup\limits_{\Vert x\Vert=1} \vert\langle Tx,x\rangle\vert.
\end{equation*}
\end{lemma}

\begin{theorem}\textbf{Diagonalisation des opérateurs auto-adjoints compacts.}
On suppose que $H$ est séparable. Soit $T\in \LL(H)$ auto-adjoint et compact, alors il existe une base hilbertienne de $H$ formée de vecteurs propres de $T$. 
\end{theorem}

\begin{exo}\textbf{$\blacktriangleright$ DEV (avec la proposition~\ref{prop:spectre} dans le cas auto-adjoint) $\blacktriangleleft$ Preuve, \cite{gourdonal} p.413}
\begin{enumerate}
\item Montrer qu'il existe une suite $(x_n)$ de vecteurs de norme 1 telle que $\ds \vert\langle Tx_n,x_n\rangle\vert \to \Vert T\Vert$.
\item Si $T\neq 0$, montrer que $T$ admet au moins une valeur propre non nulle.
\item Montrer que les sous espaces propres de $T$ sont deux à deux orthogonaux.
\item On note $G$ le sous espace engendré par tous les sous espaces propres associés à des valeurs propres non nulles et $F=\overline G$. Montrer que $F^{\perp}=\ker T$.
\item Conclure
\end{enumerate}
\end{exo}


\subsection{Retour sur $H^1_0(I)$, \cite{brezis}}


\begin{prop}\textbf{Compacité de l'injection de $H^1(I)$ dans $\CC(\overline I)$, \cite{brezis} p.129.} 
\label{prop:inj_compacte}
L'injection 
\begin{equation*}
i : \left\{
\begin{aligned}
H^1(I) & \to \CC(\overline I)\\
u & \mapsto u 
\end{aligned}
\right.
\end{equation*}
est compacte.
\end{prop}

\begin{exo}\textbf{Preuve.}
Prouver ce théorème \textit{(on pourra penser au théorème d'Ascoli, cf TD de topologie)}.
\end{exo}

\begin{theorem}\textbf{Une base hilbertienne adaptée au problème \eqref{eq:dirichlet_homo}, \cite{brezis} p.145 et p.192.}
Il existe $\left(\lambda_n\right)_{n\geq 1}$ une suite de réels strictement positifs et tendant vers $+\infty$ et $\left(e_n\right)_{n\geq 1}$ une base hilbertienne de $L^2(I)$ telle que $e_n\in \CC^2(\overline I)$ et
\begin{equation*}
\left\{
\begin{aligned}
& -e_n''+e_n=\lambda_n e_n \\
& e_n(0)=0=e_n(1),
\end{aligned}
\right.
\quad \forall n\geq 1.
\end{equation*}
De plus, $\ds \left(\tilde e_n=\frac{e_n}{\sqrt{\lambda_n-1}}\right)_{n\geq 1}$ est une base hilbertienne de $H^1_0(I)$ pour le produit scalaire $\langle u,v\rangle_{H^1_0(I)} = \langle u',v'\rangle_{L^2(I)}$.
\end{theorem}

\begin{rem}
Soit $f\in L^2$. Si on connait sa décomposition dans la base hilbertienne $(e_n)$ : $\ds f=\sum_{n=1}^{\infty} f_n e_n$, on peut construire explicitement une solution de \eqref{eq:dirichlet_homo} par $\ds u=\sum_{n=1}^{\infty} \frac{f_n}{\lambda_n} e_n$.
\end{rem}

\begin{exo}\textbf{Preuve du théorème.}
\begin{enumerate}
\item On considère $\tilde T : L^2(I)\to H^1_0(I)$ qui à $f$ associe la solution de \eqref{eq:dirichlet_homo}, et $T=i\circ \tilde T$, où $i$ est l'injection canonique de $H^1_0(I)$ dans $L^2(I)$. Montrer que $T$ est un opérateur auto-adjoint compact de $L^2$ dans $L^2$.
\item En déduire l'existence des suites $(e_n)$ et $(\lambda_n)$ satisfaisant les propriétés du théorème.
\item Dans notre cas ($I=]0,1[$), calculer explicitement $(e_n)$ et $(\lambda_n)$.
\end{enumerate}
\end{exo}

\begin{rem}
\begin{enumerate}
\item Les éléments propres obtenus dépendent des conditions aux limites (changer les conditions aux limites revient à changer l'opérateur $\tilde T$ et donc possiblement ses éléments propres), pour s'en convaincre on pourra calculer explicitement les $(e_n)$ et $(\lambda_n)$ en remplaçant la condition de Dirichlet homogène ($e_n(0)=0=e_n(1)$) par une condition de Neumann homogène : $e'_n(0)=0=e'_n(1)$.
\item Le fait qu'on considère ici un intervalle borné est essentiel, l'injection de $H^1_0(I)$ dans $L^2(I)$ n'est pas compacte sinon.
\item Ce type de résultat se généralise à des problèmes plus généraux et en dimension supérieure du type
\begin{equation*}
-\Div (A(x)\nabla u) + c(x)u = 0,
\end{equation*}
où $A$ vérifie la condition dite d'ellipticité :
\begin{equation*}
\exists a_0>0, \forall x\in I, \forall \xi \in\R^n,\quad {}^t\xi A(x)\xi \geq a_0 \left\vert \xi\right\vert^2.
\end{equation*}
\end{enumerate}
\end{rem}
